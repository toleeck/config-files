#!/bin/sh

#link to iptv-playlist
URL="http://list.satfor.pro/tv-m3u/toleeck-l5eO4dXtOq"
#temp for playlist
M3U="/tmp/playlist.m3u"
#location for a new playlist with ffmpeg pipe
M3U_FFMPEG="/home/hts/.hts/tvheadend/m3u/sharaFFMPEG.m3u"

#search text. there will be a pipe in this line.
STRING="ott-n.shara.club"
#start of ffmpeg pipe
START="pipe:\/\/\/usr\/bin\/ffmpeg -loglevel quiet -i "
#end of ffmpeg pipe
END=" -c copy -tune zerolatency -flags +global_header -strict -2 -f mpegts pipe:1"


wget -O $M3U_SHARA $URL
sed "/${STRING}/s/^/${START}/;/${STRING}/s/$/${END}/;s/\r//g" $M3U_SHARA > $M3U_SHARA_FFMPEG

chown hts:video $M3U_SHARA_FFMPEG
